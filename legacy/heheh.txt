@dataclass(frozen=True)
class ModelInfo:
    name: str

    size: int

class StopRequested(Exception):

class OllamaClient:
    def __init__(self, base_url: str) -> None:
        self.base_url = base_url.rstrip("/")

    def _post_json(self, endpoint: str, payload: dict[str, Any]) -> dict[str, Any]:
        url = f"{self.base_url}{endpoint}"
        data = json.dumps(payload).encode("utf-8")
        req = urllib.request.Request(
            url,
            data=data,
            headers={"Content-Type": "application/json"},
            method="POST",
        )
        with urllib.request.urlopen(req, timeout=120) as resp:
            return json.loads(resp.read().decode("utf-8"))

    def _get_json(self, endpoint: str) -> dict[str, Any]:
        url = f"{self.base_url}{endpoint}"
        with urllib.request.urlopen(url, timeout=30) as resp:
            return json.loads(resp.read().decode("utf-8"))

    def list_models(self) -> list[ModelInfo]:
        payload = self._get_json("/api/tags")
        out: list[ModelInfo] = []
        for m in payload.get("models", []):
            out.append(
                ModelInfo(
                    name=m.get("name", ""),
                    digest=m.get("digest", ""),
                    size=int(m.get("size", 0) or 0),
                    modified_at=m.get("modified_at", ""),
                )
            )
        return [m for m in out if m.name]

    def generate(
        self,
        model: str,
        prompt: str,
        seed: int,
        temperature: float,
        top_k: int,
        top_p: float,
        num_predict: int,
    ) -> dict[str, Any]:
        payload = {
            "model": model,
            "prompt": prompt,
            "stream": False,
            "options": {
                "seed": seed,
                "temperature": temperature,
                "top_k": top_k,
                "top_p": top_p,
                "num_predict": num_predict,
            },
        }
        return self._post_json("/api/generate", payload)


def utc_now() -> str:
    return dt.datetime.now(dt.timezone.utc).isoformat()


def slugify(text: str) -> str:
    text = text.strip().lower()
    text = re.sub(r"[^a-z0-9._-]+", "_", text)
    return text.strip("_") or "unnamed"


def stable_text_embedding(text: str, dims: int = 64) -> np.ndarray:
    """
    Deterministic, dependency-light embedding.
    Uses repeated SHA-256 bytes to build a fixed vector.
    """
    digest = hashlib.sha256(text.encode("utf-8", errors="ignore")).digest()
    needed = dims
    buf = bytearray()
    counter = 0
    while len(buf) < needed:
        block = hashlib.sha256(digest + counter.to_bytes(4, "little")).digest()
        buf.extend(block)
        counter += 1

    arr = np.frombuffer(bytes(buf[:needed]), dtype=np.uint8).astype(np.float32)
    arr = (arr - 127.5) / 127.5
    norm = np.linalg.norm(arr)
    if norm > 0:
        arr = arr / norm
    return arr


def pca_2d(matrix: np.ndarray) -> np.ndarray:
    """Project vectors to 2D via PCA (SVD)."""
    if matrix.shape[0] < 2:
        return np.zeros((matrix.shape[0], 2), dtype=np.float32)
    centered = matrix - matrix.mean(axis=0, keepdims=True)
    _, _, vt = np.linalg.svd(centered, full_matrices=False)
    pcs = vt[:2].T
    proj = centered @ pcs
    if proj.shape[1] < 2:
        pad = np.zeros((proj.shape[0], 2 - proj.shape[1]), dtype=proj.dtype)
        proj = np.hstack([proj, pad])
    return proj.astype(np.float32)


def write_json(path: pathlib.Path, payload: dict[str, Any]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("w", encoding="utf-8") as f:
        json.dump(payload, f, indent=2, ensure_ascii=False)


def append_ndjson(path: pathlib.Path, record: dict[str, Any]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("a", encoding="utf-8") as f:
        f.write(json.dumps(record, ensure_ascii=False) + "\n")


def generate_map_artifacts(
    run_dir: pathlib.Path,
    model_info: ModelInfo,
    model_records: list[dict[str, Any]],
    trajectories_per_model: int,
) -> None:
    vectors = np.vstack([stable_text_embedding(r["response_text"]) for r in model_records])
    points = pca_2d(vectors)

    model_slug = slugify(model_info.name)
    map_csv = run_dir / "maps" / f"{model_slug}_map.csv"
    map_json = run_dir / "maps" / f"{model_slug}_map_summary.json"

    map_csv.parent.mkdir(parents=True, exist_ok=True)
    with map_csv.open("w", encoding="utf-8", newline="") as f:
        writer = csv.writer(f)
        writer.writerow(
            [
                "trajectory_index",
                "model",
                "prompt_index",
                "prompt_text",
                "seed",
                "response_sha256",
                "x",
                "y",
            ]
        )
        for rec, xy in zip(model_records, points, strict=True):
            writer.writerow(
                [
                    rec["trajectory_index"],
                    rec["model"],
                    rec["prompt_index"],
                    rec["prompt_text"],
                    rec["seed"],
                    rec["response_sha256"],
                    float(xy[0]),
                    float(xy[1]),
                ]
            )

    summary = {
        "generated_at_utc": utc_now(),
        "model": {
            "name": model_info.name,
            "digest": model_info.digest,
            "size": model_info.size,
            "modified_at": model_info.modified_at,
        },
        "trajectory_count": len(model_records),
        "required_trajectory_count": trajectories_per_model,
        "map_method": "deterministic_hash_embedding_64d + pca_2d",
        "centroid": {
            "x": float(points[:, 0].mean()) if len(points) else 0.0,
            "y": float(points[:, 1].mean()) if len(points) else 0.0,
        },
        "spread": {
            "x_std": float(points[:, 0].std()) if len(points) else 0.0,
            "y_std": float(points[:, 1].std()) if len(points) else 0.0,
        },
        "map_csv": str(map_csv.relative_to(run_dir)),
    }
    write_json(map_json, summary)


class Runner:
    def __init__(self, args: argparse.Namespace) -> None:
        self.args = args
        self.client = OllamaClient(args.ollama_url)
        self.stop_requested = False

        self.run_id = dt.datetime.now(dt.timezone.utc).strftime("run_%Y%m%dT%H%M%SZ")
        self.run_dir = pathlib.Path(args.output_dir) / self.run_id
        self.raw_dir = self.run_dir / "raw"

        self.global_counter = 0
        self.model_rotation_index = 0

    def request_stop(self, signum: int, _frame: Any) -> None:
        print(f"\n[signal] Received {signum}. Finishing current call, then stopping...")
        self.stop_requested = True

    def _next_seed(self) -> int:
        self.global_counter += 1
        return self.args.seed_base + self.global_counter

    def _select_prompt(self, idx_within_model: int) -> tuple[int, str]:
        # Deterministic round-robin prompt scheduling for balance.
        prompt_index = idx_within_model % len(PROMPTS)
        return prompt_index, PROMPTS[prompt_index]

    def _write_run_manifest(self) -> None:
        manifest = {
            "run_id": self.run_id,
            "started_at_utc": utc_now(),
            "ollama_url": self.args.ollama_url,
            "trajectories_per_model": self.args.trajectories_per_model,
            "seed_base": self.args.seed_base,
            "options": {
                "temperature": self.args.temperature,
                "top_k": self.args.top_k,
                "top_p": self.args.top_p,
                "num_predict": self.args.num_predict,
            },
            "prompts": PROMPTS,
            "note": "Map artifacts are generated only when a model reaches full trajectories_per_model.",
        }
        write_json(self.run_dir / "run_manifest.json", manifest)

    def _fetch_models(self) -> list[ModelInfo]:
        models = self.client.list_models()
        if self.args.models:
            allow = {m.strip() for m in self.args.models.split(",") if m.strip()}
            models = [m for m in models if m.name in allow]
        return models

    def _choose_model(self, models: list[ModelInfo]) -> ModelInfo:
        if not models:
            raise RuntimeError("No models available from Ollama.")
        model = models[self.model_rotation_index % len(models)]
        self.model_rotation_index += 1
        return model

    def _run_one_model(self, model_info: ModelInfo) -> None:
        model_slug = slugify(model_info.name)
        raw_path = self.raw_dir / f"{model_slug}.ndjson"
        model_manifest_path = self.run_dir / "models" / f"{model_slug}_manifest.json"

        model_manifest = {
            "model": {
                "name": model_info.name,
                "digest": model_info.digest,
                "size": model_info.size,
                "modified_at": model_info.modified_at,
            },
            "started_at_utc": utc_now(),
            "target_trajectories": self.args.trajectories_per_model,
            "status": "running",
        }
        write_json(model_manifest_path, model_manifest)

        print(
            f"\n[model] {model_info.name} | digest={model_info.digest[:12]}... | "
            f"target={self.args.trajectories_per_model}"
        )

        records: list[dict[str, Any]] = []
        for i in range(self.args.trajectories_per_model):
            if self.stop_requested:
                model_manifest["status"] = "stopped_early"
                model_manifest["completed_trajectories"] = len(records)
                model_manifest["ended_at_utc"] = utc_now()
                write_json(model_manifest_path, model_manifest)
                print(f"[model] stopped early at {len(records)} trajectories (no map generated).")
                raise StopRequested()

            prompt_idx, prompt = self._select_prompt(i)
            seed = self._next_seed()
            started = time.time()
            try:
                resp = self.client.generate(
                    model=model_info.name,
                    prompt=prompt,
                    seed=seed,
                    temperature=self.args.temperature,
                    top_k=self.args.top_k,
                    top_p=self.args.top_p,
                    num_predict=self.args.num_predict,
                )
                response_text = resp.get("response", "")
                done_reason = resp.get("done_reason", "")
                total_duration = int(resp.get("total_duration", 0) or 0)
            except (urllib.error.URLError, TimeoutError) as e:
                response_text = ""
                done_reason = f"error:{type(e).__name__}"
                total_duration = 0

            elapsed_ms = int((time.time() - started) * 1000)
            rec = {
                "run_id": self.run_id,
                "timestamp_utc": utc_now(),
                "trajectory_index": i,
                "model": model_info.name,
                "model_digest": model_info.digest,
                "prompt_index": prompt_idx,
                "prompt_text": prompt,
                "seed": seed,
                "response_text": response_text,
                "response_sha256": hashlib.sha256(response_text.encode("utf-8", errors="ignore")).hexdigest(),
                "done_reason": done_reason,
                "elapsed_ms": elapsed_ms,
                "ollama_total_duration_ns": total_duration,
            }
            append_ndjson(raw_path, rec)
            records.append(rec)

            if (i + 1) % self.args.progress_every == 0 or i == 0:
                print(f"  - {i + 1}/{self.args.trajectories_per_model}")

            if self.args.sleep_ms > 0:
                time.sleep(self.args.sleep_ms / 1000.0)

        # Reaching here means full target met -> generate map artifacts.
        generate_map_artifacts(
            run_dir=self.run_dir,
            model_info=model_info,
            model_records=records,
            trajectories_per_model=self.args.trajectories_per_model,
        )

        model_manifest["status"] = "completed"
        model_manifest["completed_trajectories"] = len(records)
        model_manifest["ended_at_utc"] = utc_now()
        model_manifest["map_generated"] = True
        write_json(model_manifest_path, model_manifest)
        print(f"[model] complete -> map generated ({len(records)} trajectories)")

    def run(self) -> None:
        self.run_dir.mkdir(parents=True, exist_ok=True)
        self._write_run_manifest()

        print(f"[run] id={self.run_id}")
        print(f"[run] output={self.run_dir}")

        signal.signal(signal.SIGINT, self.request_stop)
        signal.signal(signal.SIGTERM, self.request_stop)

        while not self.stop_requested:
            models = self._fetch_models()
            if not models:
                print("[warn] No models available. Retrying in 5s...")
                time.sleep(5)
                continue

            model_info = self._choose_model(models)
            try:
                self._run_one_model(model_info)
            except StopRequested:
                break
            except Exception as e:
                print(f"[error] model run failed for {model_info.name}: {e}")
                # Continue to next model; never discard already logged data.
                time.sleep(2)

        end_payload = {
            "run_id": self.run_id,
            "ended_at_utc": utc_now(),
            "stop_requested": self.stop_requested,
            "total_trajectories_attempted": self.global_counter,
        }
        write_json(self.run_dir / "run_end.json", end_payload)
        print("[run] stopped cleanly.")


def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(description="Idle topography runner for Ollama models.")
    p.add_argument("--ollama-url", default="http://localhost:11434", help="Base URL for local Ollama server.")
    p.add_argument("--output-dir", default="prompt theator/research/idle_runs", help="Directory for artifacts.")
    p.add_argument("--models", default="", help="Optional comma-separated allowlist of model names.")
    p.add_argument("--trajectories-per-model", type=int, default=1000, help="Target trajectories per model before rotating.")
    p.add_argument("--num-predict", type=int, default=96, help="Ollama num_predict.")
    p.add_argument("--temperature", type=float, default=0.8, help="Sampling temperature.")
    p.add_argument("--top-k", type=int, default=50, help="Sampling top_k.")
    p.add_argument("--top-p", type=float, default=0.95, help="Sampling top_p.")
    p.add_argument("--seed-base", type=int, default=42, help="Base seed for deterministic seed sequence.")
    p.add_argument("--sleep-ms", type=int, default=0, help="Optional delay between trajectories.")
    p.add_argument("--progress-every", type=int, default=25, help="Print progress every N trajectories.")
    return p


def main() -> int:
    args = build_parser().parse_args()
    if args.trajectories_per_model <= 0:
        print("--trajectories-per-model must be > 0", file=sys.stderr)
        return 2
    if args.progress_every <= 0:
        print("--progress-every must be > 0", file=sys.stderr)
        return 2

    runner = Runner(args)
    runner.run()
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
