import asyncio
import numpy as np
import ollama
import subprocess

def model_exists(name: str) -> bool:
    try:
        out = subprocess.check_output(["ollama", "list"]).decode()
        return name in out
    except Exception:
        return False

from typing import List, Dict
from sklearn.metrics.pairwise import cosine_similarity

class AICOStorm:
    def __init__(self, model: str = "llama3", num_trajectories: int = 5, drift_threshold: float = 0.8):
        self.model = model
        self.num_trajectories = num_trajectories
        self.drift_threshold = drift_threshold
        self.trajectories: List[str] = []
        self.embeddings: List[np.ndarray] = []
        if not model_exists(model):
            raise ValueError(
                f"Model '{model}' is not installed.\n"
                f"Run:   ollama pull {model}\n"
                f"Or choose one from: ollama list"
            )

    async def _fetch_trajectory(self, prompt: str, seed: int):
        """Fetches a single inference trajectory from Ollama."""
        options = {
            "seed": seed,
            "temperature": 0.5 + (seed % 10) / 10.0,
            "top_p": 0.9
        }
        response = ollama.generate(model=self.model, prompt=prompt, options=options)
        return response['response']

    async def _get_embedding(self, text: str):
        """Fetches embedding for a given text from Ollama."""
        response = ollama.embeddings(model=self.model, prompt=text)
        return np.array(response['embedding'])

    def calculate_dcx_divergence(self, embeddings: List[np.ndarray]) -> np.ndarray:
        """
        Calculates the Divergence score for each trajectory.
        Divergence = 1 - Cosine Similarity to the group mean.
        """
        if not embeddings:
            return np.array([])
        
        mean_embedding = np.mean(embeddings, axis=0).reshape(1, -1)
        similarities = cosine_similarity(embeddings, mean_embedding).flatten()
        return 1 - similarities

    async def run_storm(self, prompt: str):
        print(f"[*] Starting Storm with {self.num_trajectories} trajectories...")
        
        # 1. Start Storm: Parallel Inference
        tasks = [self._fetch_trajectory(prompt, seed=i*100) for i in range(self.num_trajectories)]
        self.trajectories = await asyncio.gather(*tasks)
        
        print(f"[*] Inference complete. Calculating DCX...")

        # 2. Get Embeddings for DCX Calculation
        embed_tasks = [self._get_embedding(t) for t in self.trajectories]
        self.embeddings = await asyncio.gather(*embed_tasks)
        
        # 3. Calculate DCX Divergence
        divergences = self.calculate_dcx_divergence(self.embeddings)
        
        # 4. Pruning
        survivors = []
        for i, div in enumerate(divergences):
            status = "CLEAN" if div < self.drift_threshold else "PRUNED"
            print(f"    [Trace {i}] Divergence: {div:.4f} | Status: {status}")
            if div < self.drift_threshold:
                survivors.append(self.trajectories[i])
        
        if not survivors:
            print("[!] Critical Failure: All trajectories pruned due to extreme drift.")
            return "Internal context snapback triggered. No coherent output produced."

        # 5. Collapse: Select the most central (lowest divergence) survivor
        best_index = np.argmin(divergences)
        print(f"[*] Storm collapsed. Best trajectory: Trace {best_index}")
        
        glyph = self.trajectories[best_index]
        return glyph

async def main():
    prompt = "Explain the architectural philosophy of a system that is 'allergic to leaks'."
    storm = AICOStorm(model="llama3", num_trajectories=5)
    result = await storm.run_storm(prompt)
    print("\n--- FINAL GLYPH ---")
    print(result)

if __name__ == "__main__":
    asyncio.run(main())
